Get-ChildItem -Path . -Recurse -Include *.js,*.jsx,*.ts,*.tsx,*.json,*.css | ForEach-Object { [PSCustomObject]@{ File = $_.FullName; Lines = (Get-Content -LiteralPath $_.FullName -ErrorAction SilentlyContinue).Count } } | Tee-Object -Variable stats | Format-Table -AutoSize; Write-Host "`nTotal Lines in All Files: $($stats | Measure-Object -Property Lines -Sum | Select-Object -ExpandProperty Sum)"
